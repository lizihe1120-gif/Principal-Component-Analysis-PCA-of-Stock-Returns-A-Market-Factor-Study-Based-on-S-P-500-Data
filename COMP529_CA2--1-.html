# Elbow finder function (distance-from-line method)
find_elbow <- function(x, y) {
  x <- (x - min(x)) / (max(x) - min(x))
  y <- (y - min(y)) / (max(y) - min(y))
  y_smooth <- stats::filter(y, rep(1/3, 3), sides = 2)
  y_smooth[is.na(y_smooth)] <- y[is.na(y_smooth)]
  p1 <- c(x[1], y_smooth[1])
  p2 <- c(x[length(x)], y_smooth[length(y_smooth)])
  dist <- abs((p2[2] - p1[2]) * x - (p2[1] - p1[1]) * y_smooth + 
              p2[1]*p1[2] - p2[2]*p1[1]) /
          sqrt((p2[2] - p1[2])^2 + (p2[1] - p1[1])^2)
  bias <- seq_along(dist)
  dist_adj <- dist / sqrt(bias)
  which.max(dist_adj)
}
